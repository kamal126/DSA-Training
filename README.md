    // Stack implementation using an array
    class Stack {
        int top;
        int maxSize;
        int* arr;

    public:
        Stack(int size) {
            maxSize = size;
            arr = new int[maxSize];
            top = -1;
        }
        void push(int x) {
            if (top < maxSize - 1) {
                arr[++top] = x;
            }
        }
        int pop() {
            if (top >= 0) {
                return arr[top--];
            }
            return -1; // Stack is empty
        }
    };
